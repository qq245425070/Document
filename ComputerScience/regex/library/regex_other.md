/**
 * 正则：电话号码
 */
String TEL = "^0\\d{2,3}[- ]?\\d{7,8}";
/**
 * 正则：身份证号码15位
 */
String ID_CARD15 = "^[1-9]\\d{7}((0\\d)|(1[0-2]))(([0|1|2]\\d)|3[0-1])\\d{3}$";
/**
 * 正则：身份证号码18位
 */
String ID_CARD18 = "^[1-9]\\d{5}[1-9]\\d{3}((0\\d)|(1[0-2]))(([0|1|2]\\d)|3[0-1])\\d{3}([0-9Xx])$";

 * 正则：邮箱

String EMAIL = "^\\w+([-+.]\\w+)*@\\w+([-.]\\w+)*\\.\\w+([-.]\\w+)*$";
/**
 * 正则：URL
 */
String URL = "[a-zA-z]+://[^\\s]*";
/**
 * 正则：汉字
 */
String ZH = "^[\\u4e00-\\u9fa5]+$";
/**
 * 正则：用户名，取值范围为a-z,A-Z,0-9,"_",汉字，不能以"_"结尾,用户名必须是6-20位
 */
String USERNAME = "^[\\w\\u4e00-\\u9fa5]{6,20}(?<!_)$";
/**
 * 正则：yyyy-MM-dd格式的日期校验，已考虑平闰年
 */
String DATE = "^(?:(?!0000)[0-9]{4}-(?:(?:0[1-9]|1[0-2])-(?:0[1-9]|1[0-9]|2[0-8])|(?:0[13-9]|1[0-2])-(?:29|30)|(?:0[13578]|1[02])-31)|(?:[0-9]{2}(?:0[48]|[2468][048]|[13579][26])|(?:0[48]|[2468][048]|[13579][26])00)-02-29)$";

String IP = "((2[0-4]\\d|25[0-5]|[01]?\\d\\d?)\\.){3}(2[0-4]\\d|25[0-5]|[01]?\\d\\d?)";

/**
 * 正则：双字节字符(包括汉字在内)
 */
String DOUBLE_BYTE_CHAR = "[^\\x00-\\xff]";
/**
 * 正则：空白行
 */
String BLANK_LINE = "\\n\\s*\\r";

String POSITIVE_INTEGER = "^[1-9]\\d*$";

String POSITIVE_FLOAT = "^[1-9]\\d*\\.\\d*|0\\.\\d*[1-9]\\d*$";
