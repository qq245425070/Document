二叉树的遍历  

###先序遍历  
```
private void preOrderAsList(Node<D> node) {
    if (node == null) {
        return;
    }
    Node<D> curr = node;
    Stack<Node<D>> stack = new Stack<>();
    while (curr != null || !stack.isEmpty()) {
        if (curr != null) {
            System.out.println(curr.data);  //  println
            stack.push(curr);
            curr = curr.left;
        } else {
            curr = stack.pop();
            curr = curr.right;
        }
    }
}
```

### 后序遍历  
```
private void postOrderAsList(Node<D> node) {
    if (node == null) {
        return;
    }
    Stack<Node<D>> stack = new Stack<>();
    stack.push(node);
    Node<D> pre = null;
    Node<D> curr = null;
    while (!stack.isEmpty()) {
        curr = stack.peek();
        if (curr.left != null && curr.left != pre && curr.right != pre) {
            stack.push(curr.left);
        } else if (curr.right != null && curr.right != pre) {
            stack.push(curr.right);
        } else if (!stack.isEmpty()) {
            stack.pop();
            System.out.println(curr.data);  //  println
            pre = curr;
        }
    }
}
```
### 层序遍历  
```
private void levelVisitAsList(Node<D> node) {
    if (node == null) {
        return;
    }
    Queue<Node<D>> queue = new LinkedList<>();
    queue.add(node);
    while (!queue.isEmpty()) {
        Node<D> curr = queue.poll();
        System.out.println(curr.data);  //  println
        if (curr.left != null) {
            queue.add(curr.left);
        }
        if (curr.right != null) {
            queue.add(curr.right);
        }
    }
}
```
