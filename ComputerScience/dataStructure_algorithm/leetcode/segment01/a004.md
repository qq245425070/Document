数组, 奇数在左, 偶数在右  
```
public static void test(int array[]) {
    if (array == null || array.length <= 1) {
        return;
    }
    int first = 0;
    for (int i = 0; i < array.length; i++) {
        //  先找到第一个 偶数, 再如果当前数, 是奇数, 把当前数, 和第一个偶数交换;
        for (int j = first; j < i; j++) {
            if (array[j] % 2 == 0) {
                first = j;
                break;
            }
        }
        if (array[i] % 2 == 1) {
            int temp = array[first];
            array[first] = array[i];
            array[i] = temp;
        }
    }
}
```
